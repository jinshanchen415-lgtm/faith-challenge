var state = {
  current: 0,
  selected: -1,
  counts: { A: 0, B: 0, C: 0 }
};

function $(id) { return document.getElementById(id); }

function startGame() {
  $("page-intro").classList.add("hidden");
  $("page-game").classList.remove("hidden");
  renderScene();
}

function renderScene() {
  var scene = DATA.scenes[state.current];
  $("game-icon").textContent = scene.icon;
  $("game-title").textContent = scene.title;
  $("game-progress").textContent = "情境 " + (state.current + 1) + " / 5";
  $("game-situation").textContent = scene.situation;

  var dotsHTML = "";
  for (var i = 0; i < 5; i++) {
    dotsHTML += "<div class='dot " + (i <= state.current ? "dot-active" : "") + "'></div>";
  }
  $("game-dots").innerHTML = dotsHTML;

  var choicesHTML = "";
  for (var j = 0; j < scene.choices.length; j++) {
    choicesHTML += "<button class='choice' data-index='" + j + "'>" + scene.choices[j].text + "</button>";
  }
  $("game-choices").innerHTML = choicesHTML;

  var btns = document.querySelectorAll(".choice");
  for (var k = 0; k < btns.length; k++) {
    btns[k].onclick = function() { selectChoice(parseInt(this.getAttribute("data-index"))); };
  }

  $("game-feedback").classList.add("hidden");
  $("btn-next").disabled = true;
  $("btn-next").className = "btn btn-disabled";
  $("btn-next").textContent = state.current < 4 ? "下一題" : "看看我的路徑";
  state.selected = -1;
}

function selectChoice(idx) {
  state.selected = idx;
  var btns = document.querySelectorAll(".choice");
  for (var i = 0; i < btns.length; i++) {
    btns[i].className = (i === idx) ? "choice choice-selected" : "choice";
  }
  var fb = DATA.scenes[state.current].choices[idx].feedback;
  $("game-feedback").innerHTML = "💭 " + fb;
  $("game-feedback").classList.remove("hidden");
  $("btn-next").disabled = false;
  $("btn-next").className = "btn btn-primary";
}

function nextScene() {
  if (state.selected >= 0) {
    var t = DATA.scenes[state.current].choices[state.selected].type;
    state.counts[t]++;
  }
  if (state.current < 4) {
    state.current++;
    renderScene();
  } else {
    showResult();
  }
}

function showResult() {
  $("page-game").classList.add("hidden");

  var maxType = "A";
  if (state.counts.B > state.counts[maxType]) maxType = "B";
  if (state.counts.C > state.counts[maxType]) maxType = "C";

  var r = DATA.paths[maxType];
  var types = ["A", "B", "C"];

  var othersHTML = "";
  for (var i = 0; i < types.length; i++) {
    var k = types[i];
    if (k !== maxType) {
      var p = DATA.paths[k];
      var short = p.desc.split("。")[0] + "。";
      othersHTML += "<div class='other-item box-" + k + "'>";

      othersHTML += "<div class='other-desc'>" + short + "</div>";
      othersHTML += "</div>";
    }
  }
